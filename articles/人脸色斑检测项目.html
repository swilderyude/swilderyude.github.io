启发（但是甲方多种光照无法实现） VISIA 图像分析原理 (医美行业标准) VISIA 是由美国 Canfield Scientific 公司开发的皮肤分析系统，是目前全球医美机构公认的检测标准。

  • 核心原理参考：Canfield Scientific - Imaging Systems

    • 借鉴点：

      1. 交叉偏振光 (Cross-Polarized Lighting)：这是检测色斑的核心技术。通过滤掉表皮的反射光，直接拍摄到真皮层和表皮深层的黑色素沉着。

      2. UV 光成像：使用紫外光激发出皮肤黑色素的对比度，使肉眼看不见的潜伏斑清晰显现。

      3. 多光谱成像：区分血红蛋白（红色区）和黑色素（褐色区）。

    • 参考网址：Canfield Scientific - VISIA Complexion Analysis

import os

def merge_all_datasets():
    # 最终合并到的目标文件夹
    target_root = 'data'
    # 需要识别的标准子集
    target_splits = ['train', 'valid', 'test']
    
    # 统计数据
    stats = {'images': 0, 'json': 0, 'labels': 0, 'skipped': 0}

    print("开始深度扫描并合并所有数据集...")

    # 使用 os.walk 递归遍历当前目录下所有的文件夹
    # 这可以确保无论数据集嵌套在几层目录下都能被找到
    for root, dirs, files in os.walk('.'):
        
        # 如果当前文件夹名字是 ['train', 'valid', 'test'] 之一
        # 且我们不在目标文件夹 'data' 内部
        dir_name = os.path.basename(root)
        if dir_name in target_splits and target_root not in root:
            
            split = dir_name # train, valid 或 test
            print(f"发现数据集片段: {root}")

            # 1. 处理 images 文件夹 (图片 + JSON)
            src_img_dir = os.path.join(root, 'images')
            if os.path.exists(src_img_dir):
                dst_img_dir = os.path.join(target_root, split, 'images')
                os.makedirs(dst_img_dir, exist_ok=True)
                
                for f in os.listdir(src_img_dir):
                    if f.startswith('.'): continue
                    src_file = os.path.join(src_img_dir, f)
                    dst_file = os.path.join(dst_img_dir, f)
                    
                    if not os.path.exists(dst_file):
                        shutil.copy2(src_file, dst_file)
                        if f.lower().endswith('.json'): stats['json'] += 1
                        else: stats['images'] += 1
                    else:
                        stats['skipped'] += 1

            if os.path.exists(src_lab_dir):

import os
import json

def process_full_dataset():
    label_dir = 'labels'
    image_dir = 'images'
    img_exts = ['.jpg', '.jpeg', '.png', '.bmp', '.webp']

    # 1. TXT 映射 (旧 ID : 新 ID) - 顺序: spot(0), acne(1), mole(2)
    txt_mapping = {
        '5': '0', # spot -> spot
        '0': '1', # acne -> acne
        '3': '1', # pimple -> acne
        '1': '2', # mole1 -> mole
        '2': '2'  # mole2 -> mole
    }

    # 2. JSON 映射 (旧类名 : 新类名)
    json_mapping = {
        'spot': 'spot',
        'acne': 'acne',
        'pimple': 'acne',
        'mole1': 'mole',
        'mole2': 'mole'
    }

    if not os.path.exists(label_dir) or not os.path.exists(image_dir):
        print("错误：请确保脚本位于 images 和 labels 文件夹同级目录下！")
        return

    txt_files = [f for f in os.listdir(label_dir) if f.endswith('.txt') and f != 'classes.txt']
    
    count_kept = 0
    count_deleted = 0

    print(f"开始同步处理 TXT 和 JSON 文件...")

    for txt_name in txt_files:
        base_name = os.path.splitext(txt_name)[0]
        txt_path = os.path.join(label_dir, txt_name)
        json_path = os.path.join(image_dir, base_name + ".json")
        
        # --- A. 处理 TXT 内容 ---
        new_txt_lines = []
        try:
            with open(txt_path, 'rb') as f:
                content = f.read().decode('utf-8', errors='ignore')
            for line in content.splitlines():
                parts = line.strip().split()
                if len(parts) >= 5 and parts[0] in txt_mapping:
                    parts[0] = txt_mapping[parts[0]]
                    new_txt_lines.append(" ".join(parts))
        except Exception as e:
            print(f"读取 TXT 出错 {txt_name}: {e}")
            continue

        # --- B. 处理 JSON 内容 (如果存在) ---
        new_json_data = None
        if os.path.exists(json_path):
            try:
                with open(json_path, 'r', encoding='utf-8') as f:
                    data = json.load(f)
                
                # 过滤并修改 shapes 列表
                new_shapes = []
                for shape in data.get('shapes', []):
                    old_label = shape.get('label')
                    if old_label in json_mapping:
                        shape['label'] = json_mapping[old_label]
                        new_shapes.append(shape)
                
                # 更新 JSON 数据
                data['shapes'] = new_shapes
                new_json_data = data
            except Exception as e:
                print(f"处理 JSON 出错 {json_path}: {e}")

        # --- C. 执行文件更新或删除 ---
        # 只要 TXT 处理后为空，就认为该图片不需要了（即使 JSON 里可能有别的）
        if not new_txt_lines:
            # 1. 删 TXT
            if os.path.exists(txt_path): os.remove(txt_path)
            # 2. 删 JSON

import os

def split_image_and_labels(image_path, label_path, output_image_folder, output_label_folder, crop_size=(640, 640), overlap_ratio=0.2, min_area_ratio=0.0001):
    """切割单张图像及其标签，解决文件名冲突及小目标过滤问题"""
    image = cv2.imread(image_path)
    if image is None: return
    h, w, _ = image.shape
    crop_h, crop_w = crop_size
    img_stem = Path(image_path).stem # 获取原文件名，防止覆盖

    with open(label_path, 'r') as f:
        labels = f.readlines()

    crop_idx = 0
    # 计算步长
    step_y = int(crop_h * (1 - overlap_ratio))
    step_x = int(crop_w * (1 - overlap_ratio))

    for y in range(0, h, step_y):
        for x in range(0, w, step_x):
            # 边界处理：如果最后一块超界，则强制靠边
            curr_y = y if y + crop_h <= h else h - crop_h
            curr_x = x if x + crop_w <= w else w - crop_w
            
            if curr_y < 0 or curr_x < 0: continue # 原图比640还小的情况

            crop_image_path = os.path.join(output_image_folder, f"{crop_filename}.jpg")
            
            crop_labels = []
            for label in labels:
                parts = label.strip().split()
                if not parts: continue
                cls = parts[0]
                xc, yc, nw, nh = map(float, parts[1:])
                
                # 像素级绝对坐标（大图）
                abs_xc, abs_yc = xc * w, yc * h
                abs_w, abs_h = nw * w, nh * h
                
                # 判断中心点是否在当前切片内 (SAHI 核心逻辑)
                if curr_x <= abs_xc < curr_x + crop_w and curr_y <= abs_yc < curr_y + crop_h:
                    # 计算在小图中的相对坐标
                    new_xc = (abs_xc - curr_x) / crop_w
                    new_yc = (abs_yc - curr_y) / crop_h
                    new_nw = abs_w / crop_w
                    new_nh = abs_h / crop_h
                    
                    # 面积过滤（对于斑点，比例要设得极低）
                    if (new_nw * new_nh) >= min_area_ratio:
                        # 限制边界在 0-1 之间
                        new_xc = max(min(new_xc, 0.999), 0.001)
                        new_yc = max(min(new_yc, 0.999), 0.001)
                        crop_labels.append(f"{cls} {new_xc:.6f} {new_yc:.6f} {new_nw:.6f} {new_nh:.6f}")
            
            # 只有当切片内有目标时才保存
            if crop_labels:
                crop = image[curr_y:curr_y+crop_h, curr_x:curr_x+crop_w]
                cv2.imwrite(crop_image_path, crop)
                crop_label_path = os.path.join(output_label_folder, f"{crop_filename}.txt")
                with open(crop_label_path, 'w') as f:
                    f.write("\n".join(crop_labels))
            
            crop_idx += 1
            # 如果已经到了边界，跳出当前行的横向循环
            if curr_x + crop_w >= w: break
        if curr_y + crop_h >= h: break

def yolo_to_xyxy(label, img_w, img_h):
    cls, xc, yc, w, h = label
    x1 = (xc - w / 2) * img_w
    y1 = (yc - h / 2) * img_h
    x2 = (xc + w / 2) * img_w
    y2 = (yc + h / 2) * img_h
    return int(cls), x1, y1, x2, y2


def xyxy_to_yolo(cls, x1, y1, x2, y2, tile_w, tile_h):
    xc = ((x1 + x2) / 2) / tile_w
    yc = ((y1 + y2) / 2) / tile_h
    w = (x2 - x1) / tile_w
    h = (y2 - y1) / tile_h
    return cls, xc, yc, w, h


def read_yolo_labels(label_path, img_w, img_h):
    boxes = []
    if not label_path.exists():
        return boxes

    with open(label_path, "r", encoding="utf-8") as f:
        lines = f.readlines()

    for line in lines:
        parts = line.strip().split()
        if len(parts) != 5:
            continue

        cls = int(float(parts[0]))
        xc = float(parts[1])
        yc = float(parts[2])
        w = float(parts[3])
        h = float(parts[4])

        boxes.append(yolo_to_xyxy((cls, xc, yc, w, h), img_w, img_h))

    return boxes


def get_start_positions(length, tile_size, stride):
    """
    生成滑窗起点，保证最后一块贴到图像边缘。
    """
    if length <= tile_size:
        return [0]

    positions = list(range(0, length - tile_size + 1, stride))

    last = length - tile_size
    if positions[-1] != last:
        positions.append(last)

    return positions


def tile_one_image(img_path, label_path, out_img_dir, out_label_dir):
    img = cv2.imread(str(img_path))
    if img is None:
        print(f"无法读取图片: {img_path}")
        return

    img_h, img_w = img.shape[:2]
    boxes = read_yolo_labels(label_path, img_w, img_h)

    x_starts = get_start_positions(img_w, tile_size, stride)
    y_starts = get_start_positions(img_h, tile_size, stride)

    for y0 in y_starts:
        for x0 in x_starts:
            x1_tile = x0
            y1_tile = y0
            x2_tile = min(x0 + tile_size, img_w)
            y2_tile = min(y0 + tile_size, img_h)

            tile = img[y1_tile:y2_tile, x1_tile:x2_tile]
            tile_h, tile_w = tile.shape[:2]

            tile_labels = []

            for cls, bx1, by1, bx2, by2 in boxes:
                # 原始框面积
                box_area = max(0, bx2 - bx1) * max(0, by2 - by1)

                # 计算与 tile 的交集
                inter_x1 = max(bx1, x1_tile)
                inter_y1 = max(by1, y1_tile)
                inter_x2 = min(bx2, x2_tile)
                inter_y2 = min(by2, y2_tile)

                inter_w = max(0, inter_x2 - inter_x1)
                inter_h = max(0, inter_y2 - inter_y1)
                inter_area = inter_w * inter_h

                if inter_area <= 0:
                    continue

                # 如果目标被切得太小，则丢弃
                if box_area > 0 and inter_area / box_area < min_area_ratio:
                    continue

                # 转换成 tile 内坐标
                new_x1 = inter_x1 - x1_tile
                new_y1 = inter_y1 - y1_tile
                new_x2 = inter_x2 - x1_tile
                new_y2 = inter_y2 - y1_tile

                # 太小的框也可以过滤
                if new_x2 - new_x1 < 2 or new_y2 - new_y1 < 2:
                    continue

                yolo_label = xyxy_to_yolo(
                    cls,
                    new_x1,
                    new_y1,
                    new_x2,
                    new_y2,
                    tile_w,
                    tile_h
                )

                tile_labels.append(yolo_label)

            # 是否保留空切图
            if len(tile_labels) ** 0:
                if not keep_empty:
                    continue

                # 简单按照文件名 hash 做一个稳定采样，避免随机导致每次不同
                key = hash(f"{img_path.stem}_{x0}_{y0}") % 10000 / 10000
                if key > empty_keep_ratio:
                    continue

            out_name = f"{img_path.stem}_x{x0}_y{y0}"
            out_img_path = out_img_dir / f"{out_name}.jpg"
            out_label_path = out_label_dir / f"{out_name}.txt"

            cv2.imwrite(str(out_img_path), tile)

            with open(out_label_path, "w", encoding="utf-8") as f:
                for cls, xc, yc, w, h in tile_labels:
                    # 防止数值越界
                    xc = min(max(xc, 0), 1)
                    yc = min(max(yc, 0), 1)
                    w = min(max(w, 0), 1)
                    h = min(max(h, 0), 1)
                    f.write(f"{cls} {xc:.6f} {yc:.6f} {w:.6f} {h:.6f}\n")

def get_start_positions(length, tile_size, stride): if length <= tile_size: return [0]

positions = list(range(0, length - tile_size + 1, stride))
last = length - tile_size

if positions[-1] != last:
    positions.append(last)

return positions

def class_aware_nms(boxes, scores, classes, iou_thres=0.5): if len(boxes) ** 0: return []

boxes_t = torch.tensor(boxes, dtype=torch.float32)
scores_t = torch.tensor(scores, dtype=torch.float32)
classes_t = torch.tensor(classes, dtype=torch.int64)

keep_all = []

for cls in classes_t.unique():
    idx = torch.where(classes_t ** cls)[0]
    keep = nms(boxes_t[idx], scores_t[idx], iou_thres)
    keep_all.extend(idx[keep].tolist())

return keep_all

def sliding_window_inference(model, img): h, w = img.shape[:2]

x_starts = get_start_positions(w, tile_size, stride)
y_starts = get_start_positions(h, tile_size, stride)

all_boxes = []
all_scores = []
all_classes = []

for y0 in y_starts:
    for x0 in x_starts:
        x1 = x0
        y1 = y0
        x2 = min(x0 + tile_size, w)
        y2 = min(y0 + tile_size, h)

        tile = img[y1:y2, x1:x2]

        results = model.predict(
            source=tile,
            imgsz=tile_size,
            conf=conf_thres,
            iou=nms_iou_thres,
            device=device,
            verbose=False
        )

        result = results[0]

        if result.boxes is None or len(result.boxes) ** 0:
            continue

        boxes = result.boxes.xyxy.cpu().numpy()
        scores = result.boxes.conf.cpu().numpy()
        classes = result.boxes.cls.cpu().numpy().astype(int)

        for box, score, cls in zip(boxes, scores, classes):
            bx1, by1, bx2, by2 = box

            bx1 += x1
            bx2 += x1
            by1 += y1
            by2 += y1

            bx1 = max(0, min(bx1, w - 1))
            by1 = max(0, min(by1, h - 1))
            bx2 = max(0, min(bx2, w - 1))
            by2 = max(0, min(by2, h - 1))

            all_boxes.append([bx1, by1, bx2, by2])
            all_scores.append(float(score))
            all_classes.append(int(cls))

keep = class_aware_nms(all_boxes, all_scores, all_classes, iou_thres=nms_iou_thres)

final_boxes = [all_boxes[i] for i in keep]
final_scores = [all_scores[i] for i in keep]
final_classes = [all_classes[i] for i in keep]

return final_boxes, final_scores, final_classes

def box_iou(box1, box2): x1 = max(box1[0], box2[0]) y1 = max(box1[1], box2[1]) x2 = min(box1[2], box2[2]) y2 = min(box1[3], box2[3])

inter = max(0, x2 - x1) * max(0, y2 - y1)

area1 = max(0, box1[2] - box1[0]) * max(0, box1[3] - box1[1])
area2 = max(0, box2[2] - box2[0]) * max(0, box2[3] - box2[1])

union = area1 + area2 - inter

if union <= 0:
    return 0.0

return inter / union

def evaluate_one_image(pred_boxes, pred_scores, pred_classes, gt_boxes, gt_classes): """ 对一张图统计每类 TP FP FN """ tp = [0] * num_classes fp = [0] * num_classes fn = [0] * num_classes

pred_indices_by_cls = {c: [] for c in range(num_classes)}
gt_indices_by_cls = {c: [] for c in range(num_classes)}

for i, c in enumerate(pred_classes):
    if c in pred_indices_by_cls:
        pred_indices_by_cls[c].append(i)

for i, c in enumerate(gt_classes):
    if c in gt_indices_by_cls:
        gt_indices_by_cls[c].append(i)

for cls in range(num_classes):
    pred_idxs = pred_indices_by_cls[cls]
    gt_idxs = gt_indices_by_cls[cls]

    # 按置信度从高到低匹配
    pred_idxs = sorted(pred_idxs, key=lambda i: pred_scores[i], reverse=True)

    matched_gt = set()

    for pi in pred_idxs:
        best_iou = 0
        best_gi = -1

        for gi in gt_idxs:
            if gi in matched_gt:
                continue

            iou = box_iou(pred_boxes[pi], gt_boxes[gi])

            if iou > best_iou:
                best_iou = iou
                best_gi = gi

        if best_iou >= match_iou_thres:
            tp[cls] += 1
            matched_gt.add(best_gi)
        else:
            fp[cls] += 1

    fn[cls] += len(gt_idxs) - len(matched_gt)

return tp, fp, fn
