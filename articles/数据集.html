处理

  1. 单个数据集筛选。将不需要的标注标签的txt和图片删除，图片上有多类，仅删除不需要的一类，保留需要的类。

  2. 所有数据集标签一致。spot acne mole。data.yaml

  3. 将数据集合并为一个，合并时类别顺序一致。先把类别的01调整好，或者不按照01进行分类，而是名称。合并仅针对txt，最后仅一个data.yaml

```python title:"alter.py"
import os
import json

def process_full_dataset():
    label_dir = 'labels'
    image_dir = 'images'
    img_exts = ['.jpg', '.jpeg', '.png', '.bmp', '.webp']

    # 1. TXT 映射 (旧 ID : 新 ID) - 顺序: spot(0), acne(1), mole(2)
    txt_mapping = {
        '5': '0', # spot -> spot
        '0': '1', # acne -> acne
        '3': '1', # pimple -> acne
        '1': '2', # mole1 -> mole
        '2': '2'  # mole2 -> mole
    }

    # 2. JSON 映射 (旧类名 : 新类名)
    json_mapping = {
        'spot': 'spot',
        'acne': 'acne',
        'pimple': 'acne',
        'mole1': 'mole',
        'mole2': 'mole'
    }

    if not os.path.exists(label_dir) or not os.path.exists(image_dir):
        print("错误：请确保脚本位于 images 和 labels 文件夹同级目录下！")
        return

    txt_files = [f for f in os.listdir(label_dir) if f.endswith('.txt') and f != 'classes.txt']
    
    count_kept = 0
    count_deleted = 0

    print(f"开始同步处理 TXT 和 JSON 文件...")

    for txt_name in txt_files:
        base_name = os.path.splitext(txt_name)[0]
        txt_path = os.path.join(label_dir, txt_name)
        json_path = os.path.join(image_dir, base_name + ".json")
        
        # --- A. 处理 TXT 内容 ---
        new_txt_lines = []
        try:
            with open(txt_path, 'rb') as f:
                content = f.read().decode('utf-8', errors='ignore')
            for line in content.splitlines():
                parts = line.strip().split()
                if len(parts) >= 5 and parts[0] in txt_mapping:
                    parts[0] = txt_mapping[parts[0]]
                    new_txt_lines.append(" ".join(parts))
        except Exception as e:
            print(f"读取 TXT 出错 {txt_name}: {e}")
            continue

        # --- B. 处理 JSON 内容 (如果存在) ---
        new_json_data = None
        if os.path.exists(json_path):
            try:
                with open(json_path, 'r', encoding='utf-8') as f:
                    data = json.load(f)
                
                # 过滤并修改 shapes 列表
                new_shapes = []
                for shape in data.get('shapes', []):
                    old_label = shape.get('label')
                    if old_label in json_mapping:
                        shape['label'] = json_mapping[old_label]
                        new_shapes.append(shape)
                
                # 更新 JSON 数据
                data['shapes'] = new_shapes
                new_json_data = data
            except Exception as e:
                print(f"处理 JSON 出错 {json_path}: {e}")

        # --- C. 执行文件更新或删除 ---
        # 只要 TXT 处理后为空，就认为该图片不需要了（即使 JSON 里可能有别的）
        if not new_txt_lines:
            # 1. 删 TXT
            if os.path.exists(txt_path): os.remove(txt_path)
            # 2. 删 JSON
            if os.path.exists(json_path): os.remove(json_path)
            # 3. 删图片
            for ext in img_exts:
                img_path = os.path.join(image_dir, base_name + ext)
                if os.path.exists(img_path):
                    os.remove(img_path)
                    break
            count_deleted += 1
        else:
            # 1. 保存修改后的 TXT
            with open(txt_path, 'w', encoding='utf-8') as f:
                f.write("\n".join(new_txt_lines) + "\n")
            
            # 2. 保存修改后的 JSON (如果原本有 JSON)
            if new_json_data is not None:
                with open(json_path, 'w', encoding='utf-8') as f:
                    json.dump(new_json_data, f, ensure_ascii=False, indent=2)
            
            count_kept += 1

    # D. 更新 classes.txt
    with open(os.path.join(label_dir, 'classes.txt'), 'w', encoding='utf-8') as f:
        f.write("spot\nacne\nmole\n")

    print("-" * 40)
    print(f"同步处理完成！")
    print(f"保留并修改: {count_kept} 组 (TXT 和 JSON 已同步)")
    print(f"删除无效文件: {count_deleted} 组")
    print(f"当前类别顺序: 0:spot, 1:acne, 2:mole")

if __name__ ** "__main__":
    process_full_dataset()

def process_and_clean_dataset():
    # 基础配置
    subfolders = ['train', 'valid', 'test']
    img_exts = ['.jpg', '.jpeg', '.png', '.bmp', '.webp']
    root_classes_file = 'classes.txt'
    
    # 映射逻辑
    old_target_idx = '2' # fore
    new_target_idx = '1' # acne
    new_label_name = 'acne'

    print("开始处理数据集...")

    # 1. 遍历子文件夹
    for sub in subfolders:
        if not os.path.exists(sub):
            continue
            
        print(f"正在清洗文件夹: {sub}")
        label_dir = os.path.join(sub, 'labels')
        image_dir = os.path.join(sub, 'images')

        if not os.path.exists(label_dir) or not os.path.exists(image_dir):
            continue

        txt_files = [f for f in os.listdir(label_dir) if f.endswith('.txt') and f != 'classes.txt']
        
        kept_count = 0
        deleted_count = 0

        for txt_name in txt_files:
            base_name = os.path.splitext(txt_name)[0]
            txt_path = os.path.join(label_dir, txt_name)
            json_path = os.path.join(image_dir, base_name + ".json")
            
            # --- 处理 TXT ---
            new_txt_lines = []
            try:
                # 使用 errors='ignore' 解决 0xb0 编码报错
                with open(txt_path, 'rb') as f:
                    content = f.read().decode('utf-8', errors='ignore')
                
                for line in content.splitlines():
                    parts = line.strip().split()
                    if len(parts) >= 5 and parts[0] ** old_target_idx:
                        parts[0] = new_target_idx
                        new_txt_lines.append(" ".join(parts))
            except Exception as e:
                print(f"读取 {txt_path} 出错: {e}")
                continue

            # --- 处理 JSON ---
            new_json_data = None
            if os.path.exists(json_path):
                try:
                    with open(json_path, 'r', encoding='utf-8') as f:
                        data = json.load(f)
                    
                    new_shapes = []
                    for shape in data.get('shapes', []):
                        if shape.get('label') ** 'fore':
                            shape['label'] = new_label_name
                            new_shapes.append(shape)
                    data['shapes'] = new_shapes
                    new_json_data = data
                except: pass

            # --- 物理清理或保存 ---
            if not new_txt_lines:
                # 删除全套文件
                if os.path.exists(txt_path): os.remove(txt_path)
                if os.path.exists(json_path): os.remove(json_path)
                for ext in img_exts:
                    img_p = os.path.join(image_dir, base_name + ext)
                    if os.path.exists(img_p):
                        os.remove(img_p)
                        break
                deleted_count += 1
            else:
                # 写入修改后的内容
                with open(txt_path, 'w', encoding='utf-8') as f:
                    f.write("\n".join(new_txt_lines) + "\n")
                if new_json_data:
                    with open(json_path, 'w', encoding='utf-8') as f:
                        json.dump(new_json_data, f, ensure_ascii=False, indent=2)
                kept_count += 1
        
        print(f"[{sub}] 完成: 保留 {kept_count}, 删除 {deleted_count}")
